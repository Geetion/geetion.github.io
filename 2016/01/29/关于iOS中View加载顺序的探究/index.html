关于iOS中View加载顺序的探究

先上探究过程源代码：

ViewController1：


class ViewController: UIViewController {

 

override func loadView() {

super.loadView()

let bt = UIButton(frame: CGRect(x: 100, y: 100, width: 100, height: 100))

bt.setTitle(“to2”, forState: UIControlState.Normal)

bt.setTitleColor(UIColor.blackColor(), forState: UIControlState.Normal)

bt.addTarget(self, action: “bt”, forControlEvents: UIControlEvents.TouchUpInside)

self.view.addSubview(bt)

print(“load1”)

}

override func viewDidLoad() {

super.viewDidLoad()

print(“didload1”)

}

 

override func viewWillAppear(animated: Bool) {

print(“willappear1”)

}

 

override func viewDidAppear(animated: Bool) {

print(“didappear1”)

}

 

override func viewWillDisappear(animated: Bool) {

print(“willdisappear1”)

}

 

override func viewDidDisappear(animated: Bool) {

print(“diddisappear1”)

}

 

func bt(){

let vc = ViewController2()

self.presentViewController(vc, animated: true, completion: nil)

}

}

 

ViewController2：


class ViewController2: UIViewController {

override func loadView() {

super.loadView()

let bt = UIButton(frame: CGRect(x: 100, y: 100, width: 100, height: 100))

bt.setTitle(“to1”, forState: UIControlState.Normal)

bt.setTitleColor(UIColor.blackColor(), forState: UIControlState.Normal)

bt.addTarget(self, action: “bt”, forControlEvents: UIControlEvents.TouchUpInside)

self.view.addSubview(bt)

self.view.backgroundColor = UIColor.whiteColor()

print(“load2”)

}

 

override func viewDidLoad() {

super.viewDidLoad()

print(“didload2”)

}

 

override func viewWillAppear(animated: Bool) {

print(“willappear2”)

}

 

override func viewDidAppear(animated: Bool) {

print(“didappear2”)

}

 

override func viewWillDisappear(animated: Bool) {

print(“willdisappear2”)

}

 

override func viewDidDisappear(animated: Bool) {

print(“diddisappear2”)

}

 

func bt(){

let vc = ViewController2()

self.dismissViewControllerAnimated(true, completion: nil)

}

}

加载好View以后输出显示：

load1

didload1

willappear1

didappear1

点击Button后显示：

load2

didload2

willdisappear1

willappear2

didappear2

diddisappear1

可以看出在是先加载第二个View加载完毕后才会进行第一个View的消失操作，并且在第二个View显示完毕后才会执行第一个View的消失完成操作

然后点击返回的第一个View的Button输出如下：

willdisappear2

willappear1

didappear1

diddisappear2

可见与第一次的结果是相同的
