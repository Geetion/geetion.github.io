iOS本地数据存储之UserDefault

首先是最简单的userDefault：

这个的存取都非常方便，只要先初始化一个userDefault


var userdefault = NSUserDefaults.standardUserDefaults()

存储一个数据


userdefault.setObject(yourData, forKey: “dataKey”)

可以根据情况使用setbool,setfloat等等，读取数据


data = userdefault.objectForKey(“dataKey”)

就这么简单，但是userDefault能存储的只有一般形式的东西，比如bool，string等等，如果需要自定义一个object就需要写遵守NScoding方法了，存储这些东西的过程就像打包一个包裹然后扔进仓库，拿出来的时候要先解包才能使用，举个栗子


class NewsItem:NSObject,NSCoding {

//自定义objetc里面的各种数据

var title = String()

var author = String()

var url = String()

 

override init() {

super.init()

}

 

//解包数据

required init?(coder aDecoder: NSCoder) {

//通过键值对的形式取出数据

self.title = aDecoder.decodeObjectForKey(“title”) as! String

self.author = aDecoder.decodeObjectForKey(“author”) as! String

self.url = aDecoder.decodeObjectForKey(“url”) as! String

}

 

//打包数据

func encodeWithCoder(aCoder: NSCoder) {

//通过键值对的形式存入数据

aCoder.encodeObject(title, forKey: “title”)

aCoder.encodeObject(author, forKey: “author”)

aCoder.encodeObject(url, forKey: “url”)

}

}
